\chapter{Kien Truc He Thong}

\section{Tong Quan}
BidNova su dung kien truc Multi-layer voi Client-Server pattern:
\begin{itemize}
    \item \textbf{Client Layer:} JavaFX GUI voi nhieu controller cho auth, bidder, seller, admin
    \item \textbf{Network Layer:} Socket communication, JSON serialization (Gson)
    \item \textbf{Service Layer:} 16+ handlers, AutoBidService, AuctionManager
    \item \textbf{Data Access Layer:} DAO pattern cho User, Auction, AutoBid, BidHistory
    \item \textbf{Database Layer:} MySQL voi 4 bang chinh
\end{itemize}

\section{Luong Xu Ly}
\begin{enumerate}
    \item Client gui JSON request den Server (Port 8888)
    \item ServerSocket accept connection spawn ClientHandler thread
    \item ClientHandler parse JSON route toi ActionHandler tuong ung
    \item Handler xu ly logic, cap nhat DB, trigger broadcast
    \item ClientHandler gui response + broadcast toi tat ca clients
    \item Client UI update qua Platform.runLater()
\end{enumerate}

\chapter{Chuc Nang Chinh}

\section{1. Authentication \& User Management}
\begin{itemize}
    \item \textbf{Login:} Username/password, BCrypt verify, JWT token generation
    \item \textbf{Register:} Validate email, hash password, save DB
    \item \textbf{Delete User:} Role-based access control (ADMIN only)
\end{itemize}

\section{2. Bidding System}
\textbf{8-step validation:} User authenticated $\to$ Auction exists $\to$ Status OPEN $\to$ Not expired $\to$ Bid $>$ highest $\to$ Increment $\geq$ min $\to$ Bid $<$ ceiling $\to$ Record history

\textbf{Price Ceiling:} Tu dong ket thuc phien khi bid dat/vuot gia tran.

\textbf{Min Bid Increment:} Kiem tra buoc gia toi thieu giua 2 bid.

\section{3. AutoBid System}
\begin{itemize}
    \item User dat: maxBid, increment, gia tran
    \item Tu dong dat gia khi co bid moi (FIFO priority)
    \item Thong minh dieu chinh increment, ton trong gia tran
    \item Tu dong ket thuc phien khi dat ceiling
\end{itemize}

\section{4. Real-time Broadcasting}
\begin{itemize}
    \item Observer pattern: AuctionSubject notify tat ca AuctionObservers
    \item Event: BID\_UPDATE, AUCTION\_FINISHED, PRODUCT\_ADDED
    \item ConcurrentHashMap luu PrintWriter clients
    \item Platform.runLater() cap nhat UI tren JavaFX thread
\end{itemize}

\section{Concurrency \& Threading}
\begin{itemize}
    \item \textbf{Multi-threading:} ServerSocket accept tao ClientHandler thread cho moi client
    \item \textbf{synchronized placeBid():} Prevent race condition tren bid update
    \item \textbf{ConcurrentHashMap:} Thread-safe luu auctions, clients, handlers
    \item \textbf{CopyOnWriteArrayList:} Thread-safe luu observers, bid history
    \item \textbf{Lost update prevention:} Synchronized blocks tren critical sections
\end{itemize}

\section{Testing}
40 unit tests: User, Auction, AutoBid, BidHistory, DAO, Service, Pattern. JUnit 5 + Maven.
