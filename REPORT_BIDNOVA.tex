\chapter{Giới Thiệu Dự Án}

\section{Tóm tắt}
BidNova là một hệ thống đấu giá trực tuyến hiện đại được xây dựng bằng Java, sử dụng kiến trúc Client-Server. Hệ thống cho phép người dùng tạo, quản lý, và tham gia các phiên đấu giá với các tính năng nâng cao như tự động đặt giá (AutoBid), giá trần (Price Ceiling), và bước giá tối thiểu (Minimum Bid Increment).

\section{Mục Tiêu Dự Án}
\begin{itemize}
    \item Xây dựng một nền tảng đấu giá trực tuyến an toàn và hiệu quả
    \item Áp dụng các design pattern: Factory, Observer, Singleton, DAO, MVC
    \item Cung cấp tính năng real-time broadcasting cho tất cả users
    \item Đảm bảo tính toàn vẹn dữ liệu và concurrency control
    \item Hỗ trợ nhiều loại sản phẩm: Phương tiện, Bất động sản, Đồ nghệ thuật, Tài sản nhà nước
\end{itemize}

\section{Phạm Vi Dự Án}
\subsection{Những gì được phát triển:}
\begin{itemize}
    \item ✅ Server Java với Multi-threading
    \item ✅ Client JavaFX GUI
    \item ✅ MySQL Database
    \item ✅ Socket-based Network Communication
    \item ✅ Autobidding System
    \item ✅ Real-time Updates
    \item ✅ 10+ Unit Tests
\end{itemize}

\chapter{Design Patterns}

\section{Factory Pattern}
\subsection{Mục Đích}
Tạo các loại Item khác nhau một cách linh hoạt mà không cần client biết chi tiết.

\subsection{Cấu Trúc}
\begin{verbatim}
Item (Abstract)
├─ Vehicle (Phương tiện)
├─ RealEstate (Bất động sản)
├─ ArtCollectible (Đồ nghệ thuật)
└─ StateProperty (Tài sản nhà nước)

ItemCreator (Interface)
├─ VehicleCreator
├─ RealEstateCreator
├─ ArtCollectibleCreator
└─ StatePropertyCreator

ItemFactoryRegistry (Singleton)
└─ createItem(category, ...) → Item
\end{verbatim}

\subsection{Lợi Ích}
\begin{itemize}
    \item ✅ Dễ thêm loại sản phẩm mới
    \item ✅ Tách biệt logic tạo object
    \item ✅ Reusable và extensible
\end{itemize}

\section{Observer Pattern}
\subsection{Mục Đích}
Broadcast real-time updates tới tất cả clients khi có thay đổi trong phiên đấu giá.

\subsection{Cấu Trúc}
\begin{verbatim}
AuctionObserver (Interface)
├─ onBidPlaced()
├─ onAuctionStatusChanged()
└─ onAutoBidTriggered()

AuctionSubject
├─ attach(observer)
├─ detach(observer)
├─ notifyBidPlaced()
├─ notifyStatusChanged()
└─ notifyAutoBidTriggered()

Implementations:
├─ ClientObserver
├─ LoggingObserver
└─ ...
\end{verbatim}

\subsection{Luồng Hoạt Động}
\begin{enumerate}
    \item User A đặt giá 105M
    \item PlaceBidHandler xử lý → Cập nhật DB
    \item Gọi subject.notifyBidPlaced(105M, "UserA")
    \item AuctionSubject gọi observer.onBidPlaced() cho tất cả observers
    \item Mỗi client nhận update → Cập nhật UI
\end{enumerate}

\section{Singleton Pattern}
\subsection{Áp Dụng}
\begin{itemize}
    \item \textbf{DatabaseConnection}: Quản lý connection pool
    \item \textbf{AuctionManager}: Quản lý auction cache trong memory
    \item \textbf{ItemFactoryRegistry}: Registry cho item creators
\end{itemize}

\section{DAO Pattern}
\subsection{Các DAO Classes}
\begin{itemize}
    \item AuctionDAO: CRUD auctions
    \item UserDAO: CRUD users
    \item BidHistoryDAO: Record bids
    \item AutoBidDAO: Manage auto-bid rules
\end{itemize}

\chapter{Tính Năng Chính}

\section{1. Authentication \& Authorization}
\subsection{Quy Trình}
\begin{enumerate}
    \item User đăng nhập (username, password)
    \item LoginHandler:
    \begin{itemize}
        \item Tìm user trong DB
        \item Verify password (BCrypt)
        \item Generate JWT token
    \end{itemize}
    \item Return token cho client
    \item Mỗi request gửi token → ClientHandler validate
\end{enumerate}

\subsection{Bảo Mật}
\begin{itemize}
    \item 🔐 BCrypt password hashing (salt rounds: 10)
    \item 🔐 JWT token validation
    \item 🔐 Token expiration
    \item 🔐 Role-based access control (USER, SELLER, ADMIN)
\end{itemize}

\section{2. Bidding System}
\subsection{Validation Flow (8 kiểm tra)}
\begin{verbatim}
PLACE_BID Request
    ↓
✓ User authenticated?
    ↓
✓ Auction exists?
    ↓
✓ Status = OPEN?
    ↓
✓ Not expired?
    ↓
✓ Bid > currentHighestBid?
    ↓
✓ Increment >= minBidIncrement? ← NEW
    ↓
✓ Bid < priceCeiling? ← NEW
    ↓
✓ Update DB + Record history
    ↓
Trigger AutoBid → Broadcast Update
\end{verbatim}

\subsection{Price Ceiling Feature}
\subsubsection{Định Nghĩa}
Giá trần là mức giá tối đa mà người bán sẽ chấp nhận. Khi có bid đạt/vượt giá trần, phiên đấu giá tự động kết thúc.

\subsubsection{Ví Dụ}
\begin{verbatim}
startPrice: 100 triệu
priceCeiling: 150 triệu
minIncrement: 1 triệu

Timeline:
- 10:00 → User A bids 105M ✓
- 10:05 → AutoBid 110M ✓
- 10:10 → User C bids 140M ✓
- 10:15 → AutoBid triggers 145M ✓
- 10:20 → User A bids 150M ✓ → CEILING REACHED
           Status: FINISHED
           Winner: User A (150M)
\end{verbatim}

\subsection{Minimum Bid Increment Feature}
\subsubsection{Định Nghĩa}
Bước giá tối thiểu là khoảng cách nhỏ nhất giữa hai bid liên tiếp.

\subsubsection{Validation}
\begin{verbatim}
if (newBid - currentHighestBid) < minBidIncrement:
    REJECT with error: "Bước giá tối thiểu là X VNĐ"
\end{verbatim}

\subsubsection{Ví Dụ}
\begin{verbatim}
currentBid: 100M
minIncrement: 2M
legalBids: [102M, 103M, 104M, ...]
illegalBids: [100.5M, 101M, 101.5M]
\end{verbatim}

\section{3. AutoBid System}
\subsection{Tính Năng}
\begin{itemize}
    \item ✅ User đặt maxBid và increment
    \item ✅ Tự động đặt giá khi có bid mới
    \item ✅ FIFO priority (người đăng ký sớm được ưu tiên)
    \item ✅ Smart increment adjustment
    \item ✅ Respect price ceiling
\end{itemize}

\subsection{Execution Flow}
\begin{verbatim}
Setup:
  User B's AutoBid: maxBid = 180M, increment = 5M, ceiling = 150M

1. User A bids 100M
   → AutoBidService: nextBid = 100 + 5 = 105M
   → 105M <= 180M ✓ → Place 105M

2. User A bids 140M
   → AutoBidService: nextBid = 140 + 5 = 145M
   → 145M <= 180M ✓ → Place 145M

3. User A bids 148M
   → AutoBidService: nextBid = 148 + 5 = 153M
   → 153M > ceiling (150M)
   → Place 150M (ceiling value) + CLOSE AUCTION
   → Status: FINISHED
\end{verbatim}

\section{4. Real-time Broadcasting}
\subsection{Mechanism}
\begin{verbatim}
ClientHandler:
  static Set<PrintWriter> clientWriters (ConcurrentHashMap)
  
broadcastAll(message):
  for (PrintWriter writer : clientWriters) {
      writer.println(message)
  }
  
Triggered by:
  - PlaceBidHandler (PLACE_BID)
  - AddProductHandler (ADD_PRODUCT)
  - UpdateProductHandler (UPDATE_PRODUCT)
  - DeleteProductHandler (DELETE_PRODUCT)
  - AutoBidService (AUTO_BID_TRIGGER)
\end{verbatim}

\subsection{Update Format}
\begin{verbatim}
{
  "action": "BID_UPDATE",
  "auctionId": "auction_123",
  "newBid": 150000000,
  "bidder": "UserA",
  "timestamp": "2025-05-31T19:30:45"
}
\end{verbatim}

\chapter{Database Schema}

\section{Tables}

\subsection{users}
\begin{verbatim}
CREATE TABLE users (
    id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    email VARCHAR(100),
    full_name VARCHAR(100),
    phone VARCHAR(20),
    gender VARCHAR(10),
    role VARCHAR(20) DEFAULT 'user'
);
\end{verbatim}

\subsection{auctions}
\begin{verbatim}
CREATE TABLE auctions (
    id VARCHAR(50) PRIMARY KEY,
    product_name VARCHAR(255) NOT NULL,
    start_price DECIMAL(15,0),
    current_highest_bid DECIMAL(15,0),
    highest_bidder_id INT,
    status VARCHAR(20),
    category VARCHAR(50),
    description TEXT,
    start_time DATETIME,
    end_time DATETIME,
    seller_id INT NOT NULL,
    image_url VARCHAR(255),
    price_ceiling DECIMAL(15,0),
    min_bid_increment DECIMAL(10,0) DEFAULT 1000,
    FOREIGN KEY (seller_id) REFERENCES users(id),
    FOREIGN KEY (highest_bidder_id) REFERENCES users(id)
);
\end{verbatim}

\subsection{auto\_bids}
\begin{verbatim}
CREATE TABLE auto_bids (
    id INT PRIMARY KEY AUTO_INCREMENT,
    auction_id VARCHAR(50) NOT NULL,
    user_id INT NOT NULL,
    max_bid DECIMAL(15,0),
    increment DECIMAL(10,0),
    is_active BOOLEAN DEFAULT true,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (auction_id) REFERENCES auctions(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);
\end{verbatim}

\subsection{bid\_history}
\begin{verbatim}
CREATE TABLE bid_history (
    id INT PRIMARY KEY AUTO_INCREMENT,
    auction_id VARCHAR(50) NOT NULL,
    user_id INT NOT NULL,
    username VARCHAR(50),
    bid_amount DECIMAL(15,0),
    bid_time DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (auction_id) REFERENCES auctions(id),
    FOREIGN KEY (user_id) REFERENCES users(id)
);
\end{verbatim}

\section{Test Scenarios}

\subsection{Scenario 1: Normal Bidding}
\begin{enumerate}
    \item User A tạo phiên đấu giá (100M-150M)
    \item User B đặt giá 105M → ✅ Accept
    \item User C đặt giá 120M → ✅ Accept
    \item User B's AutoBid trigger 125M → ✅ Accept
    \item Hết thời gian → Status = FINISHED
    \item Winner: User B (125M)
\end{enumerate}

\subsection{Scenario 2: Price Ceiling Reached}
\begin{enumerate}
    \item Ceiling = 150M
    \item User A bids 148M → ✅ Accept
    \item User B's AutoBid calculates 153M > 150M
    \item System places bid at ceiling (150M)
    \item Status = FINISHED (auto-close)
    \item Winner: User B (150M)
\end{enumerate}

\subsection{Scenario 3: Invalid Bid Increment}
\begin{enumerate}
    \item Current bid: 100M
    \item Min increment: 2M
    \item User tries to bid 101M (increment = 1M)
    \item ❌ Rejected: "Bước giá tối thiểu là 2,000,000 VNĐ"
\end{enumerate}

\subsection{Scenario 4: Anti-Snipping}
\begin{enumerate}
    \item Auction ends in 2 minutes
    \item User A bids with 1 minute left
    \item System extends auction by 5 minutes
    \item New end time = original + 5 min
\end{enumerate}
